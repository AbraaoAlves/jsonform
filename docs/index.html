getJSON: function(byId) {
            var json = {}, self = this, config;
            if(typeof byId === 'object') {
                config = byId;
                byId = (config.byId)? config.byId : false;
            }
            self.find(&quot;input,textarea,select&quot;).each( function(index, item) {
                var name = (byId) ? $.trim(item.id) : $.trim(item.name);

populate: function(json, byId) {
            var self = this;
            (function roam(el, father) {
                for(var property in el) {
                    if(el[property] || el[property] === 0) {
                        var value = el[property];
                        if( typeof value == &quot;object&quot;) {
                            var parent = (!father)? property : father + &quot;\\.&quot; + property;
                            if($.isArray(value)) {
                                for(var item in value) {
                                    if(value[item]) {
                                        parent_arr = parent + &quot;\\[&quot;+item+&quot;\\]&quot;;
                                        roam(value[item], parent_arr);
                                    }
                                }
                            } else {
                                roam(value, parent);
                            }
                            parent = null;
                        } else {
                            var name = (father)? father + &quot;\\.&quot; + property : property;
                            var query = &quot;[name='" + name + "']&quot;;
                            if(byId) { query = (&quot;#&quot; + name); }
                            self.find(query).val(value);
                        }
                    }
                }
            })(json);
            return this;
        },

buildJson: function(id, valor, config) {
            
            var prefix = (config &amp;&amp; config.prefix)? config.prefix:&quot;&quot;;
            var suffix = (config &amp;&amp; config.suffix)? config.suffix:&quot;&quot;;
            
            var verifyArray = function(name, value) {
                var position;
                if( position = name.match(/\[(\d+)\]/)) {
                    var arr = [];
                    arr[ position[1] ] = value;
                    return arr;
                } else {
                    return value;
                }
            }

            return (function generateJSON(json, hierarchy, value) {
                var hierarchy = hierarchy.split(&quot;.&quot;);
                var first = hierarchy.shift();
                return function(json, name){
                    json[ prefix + name.replace(/\[(\d+)\]/, &quot;&quot;) + suffix ] = (hierarchy.length &gt; 0 ) ? 
                        verifyArray(name, generateJSON({}, hierarchy.join(&quot;.&quot;), value)) : 
                        verifyArray(name, value);
                    return json;
                }(json, first);
            })({}, id, valor);
        },
